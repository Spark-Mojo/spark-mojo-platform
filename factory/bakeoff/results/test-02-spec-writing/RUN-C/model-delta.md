def _frappe_headers() -> dict[str, str]:

def _site_header(site_name: str) -> dict[str, str]:
    """
    Add X-Frappe-Site-Name to the Frappe request headers so the multi-site
    router dispatches to the correct tenant database.
    """
    return {**_frappe_headers(), "X-Frappe-Site-Name": site_name}


async def _assert_user_exists(user_email: str, site_name: str) -> None:
    """
    Raise HTTP 404 if no Frappe User with this email exists on the given site.
    Frappe's User DocType uses email as the document name.
    """
    url = f"/api/resource/User/{user_email}"
    async with httpx.AsyncClient(
        base_url=FRAPPE_URL, headers=_site_header(site_name)
    ) as client:
        resp = await client.get(url, timeout=10)
        if resp.status_code == 404:
            raise HTTPException(
                status_code=404,
                detail=f"User '{user_email}' does not exist on site '{site_name}'.",
            )
        resp.raise_for_status()


async def _get_site_prefs(site_name: str) -> dict[str, Any]:
    """
    Read SM Site Registry for the given site and extract the
    'notification_preferences' key from config_json.
    Returns an empty dict if no site override is configured.
    """
    # SM Site Registry is named by site name (Frappe naming convention).
    url = f"/api/resource/SM Site Registry/{site_name}"
    async with httpx.AsyncClient(
        base_url=FRAPPE_URL, headers=_site_header(site_name)
    ) as client:
        resp = await client.get(url, timeout=10)
        if resp.status_code == 404:
            logger.warning(
                "SM Site Registry not found for site '%s'; "
                "site-level notification prefs will be empty.",
                site_name,
            )
            return {}
        resp.raise_for_status()
        doc = resp.json().get("data", {})

    raw_config = doc.get("config_json") or "{}"
    try:
        config = json.loads(raw_config)
    except json.JSONDecodeError:
        logger.error(
            "SM Site Registry config_json for site '%s' is not valid JSON.", site_name
        )
        return {}

    return config.get("notification_preferences", {})


async def _get_user_prefs(user_email: str, site_name: str) -> dict[str, Any]:
    """
    Read SM User Notification Preferences for the given email on the given site.
    Returns an empty dict if no user override exists.

def _merge_prefs(
    site_prefs: dict[str, Any],
    user_prefs: dict[str, Any],
) -> dict[str, ResolvedEventPrefs]:
    """
    Apply three-tier cascade and annotate each value with its authoritative tier.

    Algorithm:
        For each (event, channel) in PLATFORM_DEFAULTS:
            Start at "platform".
            If site_prefs[event][channel] exists → override, mark "site".
            If user_prefs[event][channel] exists → override, mark "user".
    """
    resolved: dict[str, ResolvedEventPrefs] = {}

    for event, channel_defaults in PLATFORM_DEFAULTS.items():
        channels: dict[str, ResolvedChannelEntry] = {}
        for channel, platform_value in channel_defaults.items():
            value = platform_value
            source = "platform"

            site_event = site_prefs.get(event, {})
            if isinstance(site_event, dict) and channel in site_event:
                site_val = site_event[channel]
                if isinstance(site_val, bool):
                    value = site_val
                    source = "site"

            user_event = user_prefs.get(event, {})
            if isinstance(user_event, dict) and channel in user_event:
                user_val = user_event[channel]
                if isinstance(user_val, bool):
                    value = user_val
                    source = "user"

            channels[channel] = ResolvedChannelEntry(value=value, source=source)

        resolved[event] = ResolvedEventPrefs(**channels)

    return resolved


def _validate_prefs_body(body: dict[str, Any]) -> list[str]:
    """
    Validate that every key in a raw preferences dict is a known event or channel.
    Returns a list of error strings (empty = valid).
    Used for PUT validation.
    """
    errors: list[str] = []
    for event_key, channel_map in body.items():
        if event_key not in VALID_EVENTS:
            errors.append(f"Unknown event key: '{event_key}'")
            continue
        if not isinstance(channel_map, dict):
            errors.append(f"Event '{event_key}' must map to an object of channel booleans.")
            continue
        for channel_key, val in channel_map.items():
            if channel_key not in VALID_CHANNELS:
                errors.append(
                    f"Unknown channel key '{channel_key}' under event '{event_key}'."
                )
            elif not isinstance(val, bool):
                errors.append(
                    f"Value for '{event_key}.{channel_key}' must be a boolean, "
                    f"got {type(val).__name__}."
                )
    return errors

@router.get(
    "/notification-preferences/{user_email}",
    response_model=NotificationPrefsResponse,
    summary="Get fully resolved notification preferences for a user",
    description=(
        "Returns the three-tier merged notification preferences for the specified user. "
        "Platform defaults are overridden by site-level overrides (from SM Site Registry "
        "config_json), which are in turn overridden by user-level overrides (from SM User "
        "Notification Preferences). Each setting is annotated with the tier that determined "
        "its final value: 'platform', 'site', or 'user'. "
        "Returns 404 if the user does not exist on the resolved site."
    ),
)
async def get_notification_preferences(
    user_email: str,
    x_frappe_site_name: str = Header(
        ...,
        alias="X-Frappe-Site-Name",
        description="The Frappe site name for the current tenant.",
    ),
) -> NotificationPrefsResponse:
    """
    GET /api/modules/admin/notification-preferences/{user_email}

    Steps:
        1. Verify X-Frappe-Site-Name header is present (FastAPI enforces this via Header(...)).
        2. Assert the user exists on the site (404 if not).
        3. Fetch site-level overrides from SM Site Registry.config_json.
        4. Fetch user-level overrides from SM User Notification Preferences.
        5. Merge all three tiers and annotate each setting with its source tier.
        6. Return resolved preferences.
    """
    # Step 2 — User existence check
    await _assert_user_exists(user_email, x_frappe_site_name)

    # Step 3 — Site prefs
    site_prefs = await _get_site_prefs(x_frappe_site_name)

    # Step 4 — User prefs
    user_prefs = await _get_user_prefs(user_email, x_frappe_site_name)

    # Step 5 — Merge
    resolved = _merge_prefs(site_prefs, user_prefs)

    return NotificationPrefsResponse(
        user_email=user_email,
        preferences=resolved,
    )

@router.put(
    "/notification-preferences/{user_email}",
    response_model=NotificationPrefsResponse,
    summary="Create or update user-level notification preferences",
    description=(
        "Accepts a partial or full preferences object and persists it as the user-level "
        "override tier in SM User Notification Preferences. Creates the document if it "
        "does not yet exist; updates it if it does. The body is merged with any existing "
        "user-level overrides (i.e. PUT is a deep-merge, not a full replacement, so that "
        "a caller setting only 'task_assigned.sms' does not wipe out previously saved "
        "settings for other events). Returns the fully resolved preferences (all three "
        "tiers) after the save. Returns 400 if any event key or channel key is invalid."
    ),
)
async def put_notification_preferences(
    user_email: str,
    body: NotificationPrefsUpdateRequest,
    x_frappe_site_name: str = Header(
        ...,
        alias="X-Frappe-Site-Name",
        description="The Frappe site name for the current tenant.",
    ),
) -> NotificationPrefsResponse:
    """
    PUT /api/modules/admin/notification-preferences/{user_email}

    Steps:
        1. Verify X-Frappe-Site-Name header is present.
        2. Assert the user exists on the site (404 if not).
        3. Validate all event/channel keys in the request body (400 if invalid).
        4. Load any existing user-level overrides from SM User Notification Preferences.
        5. Deep-merge the incoming partial body into the existing user overrides.
        6. Create or update SM User Notification Preferences in Frappe.
        7. Load site-level overrides.
        8. Merge all three tiers.
        9. Return resolved preferences.
    """
    # Step 2 — User existence check
    await _assert_user_exists(user_email, x_frappe_site_name)

    # Step 3 — Validate incoming body keys
    # Convert the Pydantic model to a plain dict, dropping None values.
    incoming_raw: dict[str, Any] = {}
    for event_key, channel_prefs in body.model_dump(exclude_none=True).items():
        if channel_prefs:  # could be empty dict after exclude_none
            incoming_raw[event_key] = channel_prefs

    validation_errors = _validate_prefs_body(incoming_raw)
    if validation_errors:
        raise HTTPException(
            status_code=400,
            detail={
                "message": "Invalid notification preference keys.",
                "errors": validation_errors,
            },
        )

    # Step 4 — Load existing user overrides (may be empty dict for a new user)
    existing_user_prefs = await _get_user_prefs(user_email, x_frappe_site_name)

    # Step 5 — Deep-merge incoming into existing.
    # Rule: incoming wins at the channel level; events not mentioned are untouched.
    merged_user_prefs: dict[str, Any] = {**existing_user_prefs}
    for event_key, channel_map in incoming_raw.items():
        if event_key not in merged_user_prefs:
            merged_user_prefs[event_key] = {}
        for channel_key, val in channel_map.items():
            merged_user_prefs[event_key][channel_key] = val

    # Step 6 — Persist to Frappe (upsert pattern).
    preferences_json_str = json.dumps(merged_user_prefs)
    doc_name_url = f"/api/resource/SM User Notification Preferences/{user_email}"

    async with httpx.AsyncClient(
        base_url=FRAPPE_URL, headers=_site_header(x_frappe_site_name)
    ) as client:
        # Attempt read to determine create vs. update.
        check_resp = await client.get(doc_name_url, timeout=10)

        if check_resp.status_code == 404:
            # CREATE
            create_resp = await client.post(
                "/api/resource/SM User Notification Preferences",
                json={
                    "doctype": "SM User Notification Preferences",
                    "user_email": user_email,
                    "preferences_json": preferences_json_str,
                },
                timeout=10,
            )
            if create_resp.status_code not in (200, 201):
                logger.error(
                    "Failed to create SM User Notification Preferences for '%s': %s",
                    user_email,
                    create_resp.text,
                )
                raise HTTPException(
                    status_code=502,
                    detail="Failed to create notification preferences in Frappe.",
                )
        else:
            # UPDATE
            check_resp.raise_for_status()
            update_resp = await client.put(
                doc_name_url,
                json={"preferences_json": preferences_json_str},
                timeout=10,
            )
            if update_resp.status_code not in (200, 201):
                logger.error(
                    "Failed to update SM User Notification Preferences for '%s': %s",
                    user_email,
                    update_resp.text,
                )
                raise HTTPException(
                    status_code=502,
                    detail="Failed to update notification preferences in Frappe.",
                )

    # Step 7 — Site prefs (for the final resolved response)
    site_prefs = await _get_site_prefs(x_frappe_site_name)

    # Step 8 — Merge all three tiers using freshly persisted user prefs
    resolved = _merge_prefs(site_prefs, merged_user_prefs)

    # Step 9 — Return resolved preferences
    return NotificationPrefsResponse(
        user_email=user_email,
        preferences=resolved,
    )

## Notes on key decisions in this spec

<details>
<summary>Why PUT is a deep-merge rather than a full replacement</summary>

A full-replacement PUT would require the caller to always send all 18 booleans to avoid silently zeroing out settings they didn't intend to touch. A React settings panel that only renders one event section at a time would constantly clobber the rest. Deep-merge (only the keys you send change) is the safer contract for a partial-preferences UI. This is called out explicitly in the docstring and acceptance criteria so the developer doesn't accidentally implement `json.dumps(incoming_raw)` directly.

</details>

<details>
<summary>Why SM User Notification Preferences uses preferences_json (Long Text) rather than child table rows</summary>

A child table would require a bench migrate every time the event or channel vocabulary changes. A JSON field lets the platform vocabulary evolve (adding a new event or channel) without a schema migration, while validation remains strict at the MAL layer via `VALID_EVENTS` and `VALID_CHANNELS`. This matches the pattern used by SM Site Registry's `config_json`.

</details>

<details>
<summary>Why the DocType is named by user_email rather than auto-name</summary>

Using `user_email` as the document name means the upsert check is a single `GET /api/resource/SM User Notification Preferences/{email}` — a deterministic 404-or-200 — with no prior list query needed. This keeps the PUT path to 2 Frappe calls in the update case and 1 in the create case.

</details>

<details>
<summary>RC-024 / ADMIN-001 dependency</summary>

The Platform Guardrails (§9, RC-024) note that admin API routes currently have zero authentication. This story spec explicitly flags ADMIN-001 as a prerequisite. If ADMIN-001 is not deployed, the developer must block this story rather than ship unauthenticated admin endpoints. No auth logic is duplicated here.

</details>
